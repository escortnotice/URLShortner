--#1  Create URL_INFO table
CREATE TABLE URL_INFO(URL_ID BIGINT(20) NOT NULL AUTO_INCREMENT, LONG_URL VARCHAR(255) NOT NULL,
SHORT_URL VARCHAR(255) NOT NULL, SHORT_URL_CREATED_TS TIMESTAMP NOT NULL, PRIMARY KEY (URL_ID),
UNIQUE KEY LONG_URL (LONG_URL), UNIQUE KEY SHORT_URL (SHORT_URL));

#2 CREATE CLICK_INFO TABLE
CREATE TABLE CLICK_INFO(URL_CLICK_ID BIGINT(20) NOT NULL AUTO_INCREMENT, URL_ID BIGINT(20), 
CLICKED_TS TIMESTAMP NOT NULL, PRIMARY KEY (URL_CLICK_ID), FOREIGN KEY (URL_ID) REFERENCES URL_ID(ID));
